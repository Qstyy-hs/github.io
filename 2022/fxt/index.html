6）检索选修四门以上课程的学生总成绩(不统计不及格的课程)，并要求按总成绩的降序排列出来 SELECT S#，SUM(GRADE) FROM SC WHERE GRADE>=60 GROUP BY S# HAVING COUNT(*) >=4 ORDER BY 2 DESC *

10）从SC表中把吴老师的女学生选课元组删去 DELETE FROM SC WHERE C# IN( SELECT C# FROM C WHERE TEACHER LIKE ‘吴%’ ) AND S# IN( SELECT S# FROM S WHERE SEX=’女’ )

11）从SC表中把数学课程中低于数学平均成绩的选课元组全部删去 DELETE FROM SC WHERE GRADE<( SELECT AVG(GRADE) FROM SC WHERE C# IN( SELECT C# FROM C WHERE CN=’数学’)) AND C# IN( SELECT C# FROM C WHERE CN=’数学’)

12）把吴老师的女同学选课成绩增加4% UPDATE SC SET GRADE=GRADE*1.04 FROM SC INNER JOIN S ON SC.S#=S.S# WHERE S.SEX=’女’ WHERE SC.C# IN( SELECT C# FROM C WHERE TEACHER LIKE ‘吴%’ ) *

SELECT DISTINCT Sname FROM S,SC WHERE S.Sno=SC.Sno AND SC.Cno IN (‘C1’,’C2’) COMPUTE COUNT(Sname)

1.查找参加唱歌队或篮球队的职工号和姓名

SELECT 职工号,姓名 FROM 职工,社会团体,参加 WHERE 职工.职工号=参加.职工号 AND 参加.编号=社会团体.编号 AND 社会团体.名称 IN (‘篮球队’,’唱歌队’);

2.查找没有参加任何社会团体的职工号 SELECT * FROM 职工 WHERE NOT EXISTS( SELECT * FROM 参加 WHERE 参加.职工号=职工.职工号 );

3.查找参加了全部社会团体的职工情况 SELECT * FROM 职工 WHERE NOT EXISTS( SELECT * FROM 社会团体 WHERE NOT EXISTS( SELECT * FROM 参加 WHERE 参加.职工号=职工.职工号 AND 参加.编号=社会团体.编号 ) );

4.求参加人数最多的社会团体的名称和参加人数 SELECT TOP 1 社会团体.名称,COUNT(参加.职工号) FROM 社会团体,参加 WHERE 社会团体.编号=参加.编号 GROUP BY 社会团体.名称 ORDER BY COUNT(参加.职工号) DESC

5.求参加人数超过100人的社会团体的名称和负责人 SELECT　社会团体.名称，职工.姓名　 FROM　职工，社会团体，参加 WHERE　社会团体.编号＝参加.编号　AND　社会团体.负责人＝职工.职工号 GROUP BY　参加.编号 HAVING COUNT(参加.编号)＞100;

6.查找参加了羽毛球社团没有参加乒乓球社团的职工号 SELECT 职工号 FROM 参加,社团 WHERE 社团.编号=参加.编号 AND 社团.名称=’羽毛球’ AND 职工号 NOT IN( SELECT 职工号 FROM 参加,社团 WHERE 社团.编号=参加.编号 AND 社团.名称=’乒乓球’)

7.求每个社团的参加人数 SELECT COUNT(职工号) FROM 参加 GROUP BY 社团编号

8.建立视图

1）社团负责人(编号，名称，负责人职工号，负责人姓名，负责人性别) CREATE VIEW 社团负责人(编号,名称,负责人职工号,负责人姓名,负责人性别) AS SELECT 编号,名称,负责人,姓名,性别 FROM 社会团体,职工 WHERE 社会团体.负责人=职工.职工号;

2)参加人情况(职工号,姓名,社团编号,社团名称,参加日期) CREATE VIEW 参加人情况(职工号,姓名,社团编号,社团名称,参加日期) AS SELECT 编号,名称,负责人,姓名,性别 FROM 职工,社会团体,参加 WHERE 职工.职工号=参加.职工号 AND 参加.编号=社会团体.编号
